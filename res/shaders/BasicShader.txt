#shader vertex
#version 330 core
layout(location = 0) in vec3 in_Position;
layout(location = 1) in vec3 in_Color;

out vec3 ex_Color;

uniform mat4 projectionMatrix;
uniform mat4 viewMatrix;



void main()
{
	vec3 diffuseLightPosition = vec3(100.f, 100.f, 0.f);
	vec3 ambientLightColour = vec3(0.6f, 0.6f, 0.6f);
	
	vec3 lightToVert = in_Position * diffuseLightPosition;

	//float dotProduct = dot(lightToVert)

	gl_Position = projectionMatrix * viewMatrix * vec4(in_Position, 1.0);

	ex_Color = ambientLightColour * in_Color;

};

#shader fragment
#version 330 core

in vec3 ex_Color;
out vec4 out_Color;

void main()
{
	out_Color = vec4(ex_Color, 1.0);
};
